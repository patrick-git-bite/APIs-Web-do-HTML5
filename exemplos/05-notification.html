Código-fonte

// 1. Solicitar permissão
Notification.requestPermission().then((permissao) => {
  console.log("Permissão:", permissao);
  // "granted", "denied" ou "default"
});

// 2. Criar e enviar notificação
if (Notification.permission === "granted") {
  const notif = new Notification("Título da Notificação", {
    body: "Corpo da mensagem aqui.",
    icon: "🔔"  // ou URL de uma imagem
  });

  // Eventos da notificação
  notif.onclick = () => {
    console.log("Usuário clicou na notificação!");
    window.focus();
  };

  notif.onclose = () => {
    console.log("Notificação fechada.");
  };
}
