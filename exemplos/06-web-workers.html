Código-fonte

// === Página principal ===

// 1. Criar o worker
const worker = new Worker("worker.js");

// 2. Enviar mensagem para o worker
worker.postMessage({ tipo: "fibonacci", dados: 42 });

// 3. Receber resultado do worker
worker.onmessage = (event) => {
  console.log("Resultado:", event.data);
};

// === Arquivo worker.js (thread separada) ===

self.onmessage = function (e) {
  const n = e.data.dados;
  const resultado = fibonacci(n);
  self.postMessage({ resultado });
};

function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}
